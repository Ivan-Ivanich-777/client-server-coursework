437 // Центрирование ASCII-строки в поле шириной width (символов, не байт)
438 static std::string center(const char *text, int width)
439 {
440 std::string s(text);
441 int len = static_cast<int>(s.size());
442 if (len >= width) return s;
443 int pad = width - len;
444 int left = pad / 2;
445 int right = pad - left;
446 return std::string(left, ' ') + s + std::string(right, ' ');
447 }
